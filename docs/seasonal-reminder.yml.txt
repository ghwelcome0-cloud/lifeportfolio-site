name: SEO Quarterly Seasonal Reminder

# Cron schedule (UTC). KST = UTC+9, so 09:00 KST = 00:00 UTC.
#
# Cron 표현식: '0 0 D M *'  → 매년 M월 D일 00:00 UTC = 09:00 KST
#
# 분기별 배포일 (KST):
#   Q1: 2026-01-08 (목)  → 1월 1일 D-7 알림 → 1월 8일 D-Day 알림
#                          12월 25일 D-14 알림 (전년)
#   Q2: 2026-05-06 (수)  → 4월 22일 D-14, 4월 29일 D-7, 5월 6일 D-Day
#   Q3: 2026-08-19 (수)  → 8월 5일 D-14, 8월 12일 D-7, 8월 19일 D-Day
#   Q4: 2026-11-18 (수)  → 11월 4일 D-14, 11월 11일 D-7, 11월 18일 D-Day
#
# LLM 인용 점검 (필러 배포 2026-05-28 기준):
#   D+7  → 2026-06-04
#   D+30 → 2026-06-27
#   D+90 → 2026-08-26
#
# 매년 같은 월·일에 반복되므로 연도는 cron에서 다루지 않음. 워크플로는
# 매년 자동 실행되며 이슈 본문에 연도 변수를 동적으로 삽입.

on:
  schedule:
    # === Q1 (1월 8일 배포) ===
    - cron: '0 0 25 12 *'   # Q1 D-14: 12-25 09:00 KST
    - cron: '0 0 1 1 *'     # Q1 D-7:  01-01 09:00 KST
    - cron: '0 1 8 1 *'     # Q1 D-Day: 01-08 10:00 KST (배포 시각)

    # === Q2 (5월 6일 배포) ===
    - cron: '0 0 22 4 *'    # Q2 D-14: 04-22 09:00 KST
    - cron: '0 0 29 4 *'    # Q2 D-7:  04-29 09:00 KST
    - cron: '0 1 6 5 *'     # Q2 D-Day: 05-06 10:00 KST

    # === Q3 (8월 19일 배포) ===
    - cron: '0 0 5 8 *'     # Q3 D-14: 08-05 09:00 KST
    - cron: '0 0 12 8 *'    # Q3 D-7:  08-12 09:00 KST
    - cron: '0 1 19 8 *'    # Q3 D-Day: 08-19 10:00 KST

    # === Q4 (11월 18일 배포) ===
    - cron: '0 0 4 11 *'    # Q4 D-14: 11-04 09:00 KST
    - cron: '0 0 11 11 *'   # Q4 D-7:  11-11 09:00 KST
    - cron: '0 1 18 11 *'   # Q4 D-Day: 11-18 10:00 KST

    # === LLM 인용 점검 (2026 기준, 연 1회) ===
    - cron: '0 0 4 6 *'     # LLM D+7:  06-04 09:00 KST
    - cron: '0 0 27 6 *'    # LLM D+30: 06-27 09:00 KST
    - cron: '0 0 26 8 *'    # LLM D+90: 08-26 09:00 KST

  # 수동 트리거 지원 (테스트용)
  workflow_dispatch:
    inputs:
      reminder_type:
        description: '알림 종류 (q1-d14, q1-d7, q1-dday, q2-d14, ..., llm-d7, llm-d30, llm-d90)'
        required: true
        default: 'q1-d14'
        type: choice
        options:
          - q1-d14
          - q1-d7
          - q1-dday
          - q2-d14
          - q2-d7
          - q2-dday
          - q3-d14
          - q3-d7
          - q3-dday
          - q4-d14
          - q4-d7
          - q4-dday
          - llm-d7
          - llm-d30
          - llm-d90

permissions:
  issues: write
  contents: read

jobs:
  create-reminder-issue:
    runs-on: ubuntu-latest
    steps:
      - name: Determine reminder type from cron schedule
        id: type
        run: |
          # 수동 실행이면 입력값 사용
          if [ "${{ github.event_name }}" = "workflow_dispatch" ]; then
            echo "reminder_type=${{ github.event.inputs.reminder_type }}" >> "$GITHUB_OUTPUT"
            exit 0
          fi

          # cron 스케줄에서 현재 날짜로 판단 (UTC 기준)
          DAY=$(date -u +%d)
          MONTH=$(date -u +%m)
          KEY="${MONTH}-${DAY}"

          case "$KEY" in
            "12-25") TYPE="q1-d14" ;;
            "01-01") TYPE="q1-d7" ;;
            "01-08") TYPE="q1-dday" ;;
            "04-22") TYPE="q2-d14" ;;
            "04-29") TYPE="q2-d7" ;;
            "05-06") TYPE="q2-dday" ;;
            "08-05") TYPE="q3-d14" ;;
            "08-12") TYPE="q3-d7" ;;
            "08-19") TYPE="q3-dday" ;;
            "11-04") TYPE="q4-d14" ;;
            "11-11") TYPE="q4-d7" ;;
            "11-18") TYPE="q4-dday" ;;
            "06-04") TYPE="llm-d7" ;;
            "06-27") TYPE="llm-d30" ;;
            "08-26") TYPE="llm-d90" ;;
            *) TYPE="unknown-$KEY" ;;
          esac

          echo "reminder_type=$TYPE" >> "$GITHUB_OUTPUT"

      - name: Compose issue title and body
        id: compose
        run: |
          TYPE="${{ steps.type.outputs.reminder_type }}"
          YEAR=$(date -u +%Y)

          # 분기별 메타 데이터 (제목, 본문, 라벨)
          case "$TYPE" in
            q1-d14)
              TITLE="[SEO] Q1 D-14 점검 — 1월 결심 글 (${YEAR}-01-08 배포)"
              BODY=$(cat <<-'EOF'

              **글 정보**:
              - 파일: `blog/posts/2026-Q1-january-resolution-fades.html` (KO)
              - 파일: `blog/posts-en/2026-Q1-january-resolution-fades.html` (EN)
              - 페르소나: 박지훈 33세, IT 사무직 6년차 (가상)

              **오늘 할 일** (체크리스트):
              - [ ] 본문 통계 최신성 확인 — 잡코리아 30대 75.3% 번아웃 / Discover Happy Habits 92%·23% / 한국경영자총협회 69.5%
              - [ ] 2025·2026 신규 보고가 있으면 dateModified 갱신
              - [ ] 페르소나 박지훈 33세 IT 사무직 — 1단계 시장조사 프로필 일치
              - [ ] 필러/시리즈 1·4 링크 깨짐 점검
              - [ ] CTA 가격 9,900원 유효성 확인

              📚 자세한 절차: [`/docs/seo-submission-guide.md`](../blob/main/docs/seo-submission-guide.md) §7
              📅 캘린더 파일: [`/docs/seo-reminders.ics`](../blob/main/docs/seo-reminders.ics)
              EOF
              )
              LABELS="seo,quarterly,q1,pre-flight"
              ;;
            q1-d7)
              TITLE="[SEO] Q1 D-7 최종검토 — 1월 결심 글 (${YEAR}-01-08 배포)"
              BODY=$(cat <<-'EOF'
              ## Q1 D-7 (1주 전 최종 검토)

              LABELS="seo,llm,monitoring,d+7"

      - name: Create reminder issue

            const labels = labelsCsv.split(',').map(s => s.trim()).filter(Boolean);

            const result = await github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: title,
              body: body,
              labels: labels
            });

            core.info(`Created issue #${result.data.number}: ${result.data.html_url}`);
