# ─────────────────────────────────────────────────────────────
#  Firebase Hosting 자동 배포
#  - main 브랜치에 머지(push)되면 자동으로 사이트에 배포됩니다.
#  - 기존 Firebase 방식 그대로 사용 (도메인·SSL·firebase.json 변경 없음).
#  - 안전장치: hosting 만 배포합니다. (functions/firestore/database 는 건드리지 않음)
#
#  ⚙️ 사전 준비 (한 번만): GitHub 저장소 Settings → Secrets and variables → Actions 에
#     아래 Secret 2개를 등록해야 합니다.  자세한 방법은 docs/firebase-auto-deploy-setup.md 참고.
#       1) FIREBASE_SERVICE_ACCOUNT  : Firebase 서비스 계정 키(JSON) 전체 내용
#       2) (선택) FIREBASE_PROJECT_ID : lifeportfolio  (기본값으로도 동작)
# ─────────────────────────────────────────────────────────────
name: Deploy to Firebase Hosting

on:
  push:
    branches:
      - main
  # 수동 실행 버튼도 제공 (Actions 탭에서 직접 돌릴 수 있음)
  workflow_dispatch:

# 동시에 여러 배포가 겹치지 않도록 (가장 최신 것만 배포)
concurrency:
  group: firebase-hosting-deploy
  cancel-in-progress: true

permissions:
  contents: read

jobs:
  deploy:
    name: Build & Deploy (hosting only)
    runs-on: ubuntu-latest
    steps:
      - name: 소스 코드 체크아웃
        uses: actions/checkout@v4

      - name: 배포 전 안전 점검 (필수 파일 존재 확인)
        run: |
          test -f firebase.json || { echo "::error::firebase.json 이 없습니다. 배포 중단."; exit 1; }
          test -f index.html   || { echo "::error::index.html 이 없습니다. 배포 중단."; exit 1; }
          echo "✅ 필수 파일 확인 완료"

      - name: Firebase Hosting 배포 (hosting 만)
        uses: FirebaseExtended/action-hosting-deploy@v0
        with:
          repoToken: ${{ secrets.GITHUB_TOKEN }}
          firebaseServiceAccount: ${{ secrets.FIREBASE_SERVICE_ACCOUNT }}
          projectId: ${{ secrets.FIREBASE_PROJECT_ID || 'lifeportfolio' }}
          channelId: live          # 실제 운영 사이트(live)로 배포
          # ⚠️ channelId: live 로 배포하면 이 액션은 hosting 만 배포합니다.
          #    functions / firestore / database 는 절대 건드리지 않습니다.
          # target 은 비워두면 firebase.json 의 기본 hosting site 로 배포됩니다.
